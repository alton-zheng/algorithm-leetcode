## 滑动窗口
```java
class Solution {

    public int maxVowels(String s, int k) {

        char[] sChar = s.toCharArray();
        int len = sChar.length;

        int[] help = new int[26];
        help[0] = help['e' - 'a'] = help['i' - 'a'] = help['o' - 'a'] = help['u' - 'a'] = 1;

        int res = 0;

        for (int i = 0; i <= k - 1; i++) {
            res += help[sChar[i] - 'a'];
        }

        int idx = 0;
        int pre = res;

        for (int i = k; i < len; i++) {
            pre += help[sChar[i] - 'a'] - help[sChar[idx] - 'a'];
            res = Math.max(res, pre);
            idx++;
        }

        return res;
    }
}
```
